SESSION         ::= MAIN_LOGON ;

MAIN_LOGON      ::= SUCCESSFUL_MAIN_LOGON
                  | FAILED_LOGIN MAIN_LOGON ; 

SUCCESSFUL_MAIN_LOGON ::= "Interactive_Logon" ACTIONS "Interactive_Logoff"
                        | "Network_Logon" "Network_Logoff";

FAILED_LOGIN    ::= "Failed_Logon" ;

ACTIONS         ::= ( ACTION )* ;

ACTION          ::= SUBLOGON
                  | OBJECT_OP
                  | REGISTRY_OP
                  | PROCESS_OP;

SUBLOGON        ::= "Batch_Logon"
                  | "RunAs_Logon"
                  | "Network_Logon" ;
                  
OBJECT_OP       ::= "Create_Object"
                  | "Delete_Object"
                  | "Modify_Object" ;

REGISTRY_OP     ::= "Create_Registry"
                  | "Delete_Registry"
                  | "Modify_Registry"
                  | "Modify_CommonStartup_Registry" ;

PROCESS_OP      ::= PROCESS_GENERIC
                  | PROCESS_CMD ;

PROCESS_GENERIC ::= "Start_Process" ;

PROCESS_CMD ::= "Start_CMD_Process" ;

SESSION         ::= MAIN_LOGON ;

MAIN_LOGON ::= "Interactive_Logon" ACTIONS "Interactive_Logoff" ;

ACTIONS         ::= ( ACTION )* ;

ACTION          ::= SUBLOGON
                  | OBJECT_OP
                  | PROCESS_OP ;

SUBLOGON        ::= "Network_Logon" ;

OBJECT_OP       ::= "Create_Object"
                  | "Delete_Object"
                  | "Modify_Object" ;

PROCESS_OP      ::= PROCESS_GENERIC
                  | PROCESS_CMD ;

PROCESS_GENERIC ::= "Start_Process" ;

PROCESS_CMD ::= "Start_CMD_Process" ;

SESSION         ::= MAIN_LOGON ;

MAIN_LOGON      ::= SUCCESSFUL_MAIN_LOGON
                  | FAILED_LOGIN MAIN_LOGON ; 

SUCCESSFUL_MAIN_LOGON ::= "Interactive_Logon" ACTIONS MAIN_LOGOFF
                        | "Network_Logon" ;

FAILED_LOGIN    ::= "Failed_Logon" ;

MAIN_LOGOFF     ::= "Logoff" ;

ACTIONS         ::= ( ACTION )* ;

ACTION          ::= SUBLOGON
                  | OBJECT_OP
                  | REGISTRY_OP
                  | PROCESS_OP
                  | LOCK_UNLOCK ;

SUBLOGON        ::= "Batch_Logon"
                  | "Service_Logon"
                  | "RunAs_Logon" ACTIONS SUBLOGOFF   // Not sure if this takes ACTIONS or more like action is just a parameter we choose
                  | "Network_Logon" ;

SUBLOGOFF       ::= "Logoff" ;

OBJECT_OP       ::= "Create_Object"
                  | "Delete_Object"
                  | "Modify_Object" ;

REGISTRY_OP     ::= "Create_Registry"
                  | "Delete_Registry"
                  | "Modify_Registry"
                  | "Modify_CommonStartup_Registry" ;

PROCESS_OP      ::= PROCESS_GENERIC
                  | PROCESS_CMD ;

PROCESS_GENERIC ::= "Start_Process" ;   // Possibly add more

PROCESS_CMD ::= "Start_CMD_Process" ;

LOCK_UNLOCK     ::= "Lock_Workstation" "Unlock_Workstation" ;
